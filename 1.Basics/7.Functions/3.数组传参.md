# 数组传参
数组作为参数传给函数，将整块内存中的数值传递给函数是不可能的，但是允许传递它的地址，而且传递地址更快速有效。

>定义数组作为参数
```c++
type func(type array[]);
```
>数组参数调用方式
```c++
func(array);
```

>Example 1: 数组作为参数
```c++
#include <stdio.h>

void printArray(int array[], int length){
    // int array[] = a;
    for(int i=0; i<length; i++)
        printf("%d ", array[i]);
    printf("\n");
}

void printArray(int (&array)[10], int length){
    // int (&array)[10] = a;
    for(int i=0; i<length; i++)
        printf("%d ", array[i]);
    printf("\n");
}

void printArray(int *array, int length){
    // int* array = a;
    for(int i=0; i<length; i++)
        printf("%d ", array[i]);
    printf("\n");
}

int main(){
    int array[10];
    for(int i=0; i<10; i++)
        array[i] = 10+i;

    printArray(&array, 10);
}
```


>Example 2: main 函数的 argv
```c++
#include <stdio.h>

void printArray(int num, int *array[]){

    int *ptr = (int*)array;
    for(int i=0; i<num; i++)
        printf("%d ", ptr[i]);
    printf("\n");
}

void printArray(int num, int **array){
    int *ptr = (int*)array;
    for(int i=0; i<num; i++)
        printf("%d ", ptr[i]);
    printf("\n");
}

// int main(int argc, char *argv[])
int main(int argc, char** argv){

    printf("输入的参数共有: %d条\n", argc);
    for (int i = 0; i < argc; i++)
        printf("%s ", argv[i]);
    printf("\n");


    int array[10];
    for(int i=0; i<10; i++)
        array[i] = 10+i;

    printArray(10, (int**)&array);

}
```
